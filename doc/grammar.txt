PROGRAM -> TRIDA PROGRAM
| eof.
TRIDA -> class identifikator lcb SEZNAM_DEFINIC_STATIC rcb.
SEZNAM_DEFINIC_STATIC -> static DATOVY_TYP DEFINICE_STATIC SEZNAM_DEFINIC_STATIC
| .
DEFINICE_STATIC -> DEFINICE_FUNKCE
| DEFINICE_PROMENNA sc.
DEFINICE_PROMENNA -> identifikator DEF_PROM_KONEC.
DEF_PROM_KONEC -> PRIRAZENI
| .
DEFINICE_FUNKCE -> f_identifikator lrb SEZNAM_PARAMETRU rrb SLOZENY_PRIKAZ.
SEZNAM_PARAMETRU -> PARAMETR_PRVNI PARAMETR_DALSI
| .
PARAMETR_PRVNI -> primitivni_typ identifikator.
PARAMETR_DALSI -> comma primitivni_typ identifikator PARAMETR_DALSI
| .
SEZNAM_VSTUPU -> vyraz VSTUP_DALSI
| identifikator VSTUP_DALSI
| .
VSTUP_DALSI -> comma VSTUP_KONEC
| .
VSTUP_KONEC -> vyraz
| identifikator.
SLOZENY_PRIKAZ -> lcb BLOK_PRIKAZU rcb.
BLOK_PRIKAZU -> PRIKAZ BLOK_PRIKAZU
| .
PRIKAZ -> primitivni_typ DEFINICE_PROMENNA sc
| vyraz sc
| identifikator POUZITI sc
| f_identifikator VOLANI_FUNKCE sc
| return NAVRAT_KONEC sc
| if lrb vyraz rrb SLOZENY_PRIKAZ else SLOZENY_PRIKAZ
| while lrb vyraz rrb SLOZENY_PRIKAZ.
POUZITI -> PRIRAZENI
| .
VOLANI_FUNKCE -> lrb SEZNAM_VSTUPU rrb.
NAVRAT_KONEC -> vyraz
| identifikator
| .
PRIRAZENI -> assign PRAVA_STRANA.
PRAVA_STRANA -> vyraz
| identifikator
| f_identifikator VOLANI_FUNKCE.
DATOVY_TYP -> void
| primitivni_typ.
